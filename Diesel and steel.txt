local JeepEnhanceTab = Window:CreateTab("ðŸš™ Jeep Enhance", 4483362458)
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer
local Camera = workspace.CurrentCamera

-- Auto Drive
local autoDriveEnabled = false
JeepEnhanceTab:CreateToggle({
    Name = "Auto Drive (towards camera)",
    CurrentValue = false,
    Callback = function(enabled)
        autoDriveEnabled = enabled
    end,
})
RunService:BindToRenderStep("AutoDrive", Enum.RenderPriority.Character.Value, function()
    if autoDriveEnabled then
        local character = LocalPlayer.Character
        if character and character:FindFirstChild("HumanoidRootPart") then
            local hrp = character.HumanoidRootPart
            local dir = Camera.CFrame.LookVector * 15 -- Velocity = 15
            hrp.Velocity = Vector3.new(dir.X, hrp.Velocity.Y, dir.Z)
        end
    end
end)

-- Infinite Fuel
local infiniteFuel = false
JeepEnhanceTab:CreateToggle({
    Name = "Infinite Fuel",
    CurrentValue = false,
    Callback = function(enabled)
        infiniteFuel = enabled
        task.spawn(function()
            while infiniteFuel do
                local character = LocalPlayer.Character
                if character then
                    local jeep = character:FindFirstChildWhichIsA("Model")
                    if jeep then
                        local fuel = jeep:FindFirstChild("Fuel")
                        if fuel and fuel:IsA("NumberValue") then
                            fuel.Value = 9999
                        end
                    end
                end
                task.wait(1)
            end
        end)
    end,
})

-- Auto Take Money + Sukli
local autoBayad = false
JeepEnhanceTab:CreateToggle({
    Name = "Auto Take Bayad + Sukli",
    CurrentValue = false,
    Callback = function(enabled)
        autoBayad = enabled
        task.spawn(function()
            while autoBayad do
                for _, gui in ipairs(game:GetDescendants()) do
                    if gui:IsA("TextButton") then
                        local text = gui.Text:lower()
                        if text:find("bayad") or text:find("sukli") or text:find("change") then
                            pcall(function() gui:Activate() end)
                        end
                    end
                end
                task.wait(0.5)
            end
        end)
    end,
})
